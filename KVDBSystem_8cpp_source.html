22 // KVDBSystem
23 //
24 // Database class used to store information on different colliding systems studied during an experiment.
25 // A 'system' is a unique combination of projectile, target and beam energy.
26 //
27 // Informations on Z & A of projectile and target and beam energy are here (GetZbeam(), GetZtarget(), etc.)
28 //
29 // GetTarget() returns a pointer to a KVTarget object describing the physical characteristics of the
30 // target for this system : such an object can be used to calculate energy losses of incident charged
31 // particles etc.
32 // NOTE: if the current run being analysed corresponds to this system, the target of INDRA will have
33 // been set using this method. i.e. you can use gIndra->GetTarget() to obtain the KVTarget object of the current
34 // run.
35 //
36 // GetKinematics() returns a pointer to a KV2Body object which can be used to obtain information on the kinematics
37 // of the colliding system. See the KV2Body class for more details.
38 // Some of the more commonly used kinematical quantities are also available directly here: GetPtot(), GetZVtot(),
39 // GetECM() etc.

124 KV2Body* KVDBSystem::GetKinematics()
125 {
126  // Create (if it doesn't already exist) and return pointer to a KV2Body object initialised
127  // with the entrance channel corresponding to this system. Use this to obtain information
128  // such as the recoil velocity of the CM, available energy, etc. (see KV2Body).
129  //
130  // If no projectile and/or target are defined for the system, we return 0x0.
131 
132  if (GetZbeam()*GetZtarget() == 0) return nullptr;
133 
134  if (!fCinema) {
135  fCinema = new KV2Body();
136  fCinema->SetProjectile(GetZbeam(), GetAbeam());
137  fCinema->SetTarget(GetZtarget(), GetAtarget());
138  fCinema->GetNucleus(1)->SetEnergy(GetEbeam() * GetAbeam());
139  fCinema->CalculateKinematics();
140  }
141  return fCinema;
142 }

229 Int_t KVDBSystem::Compare(const TObject* obj) const
230 {
231  //Function used to sort lists of systems.
232  //Systems are sorted according to the number of the first run in the
233  //(sorted) list of runs associated to the system.
234  //Systems with lower first run numbers appear earlier in the list.
235 
236  if (!GetRuns())
237  return 0;
238  KVDBSystem* other_sys =
239  dynamic_cast < KVDBSystem* >(const_cast < TObject* >(obj));
240  if (!other_sys)
241  return 0;
242  KVList* other_runs;
243  if (!(other_runs = other_sys->GetRuns()))
244  return 0;
245  Int_t first = ((KVDBRecord*) fRunlist->At(0))->GetNumber();
246  Int_t other_first = ((KVDBRecord*) other_runs->At(0))->GetNumber();
247  return (first == other_first ? 0 : (other_first > first ? -1 : 1));
248 }

304 {
305  //Write informations on system in the format used in Systems.dat files:
306  //
307  //+155Gd + 238U 36 MeV/A '+' followed by name of system
308  //155 64 238 92 36.0 Aproj Zproj Atarg Ztarg Ebeam
309  //Target: 3 0.0 target with 3 layers, angle 0 degrees
310  //C 0.02 1st layer : carbon, 20 g/cm2
311  //238U 0.1 2nd layer : uranium-238, 100 g/cm2
312  //C 0.023 3rd layer : carbon, 23 g/cm2

314 
315  f << "+" << GetName() << endl;
316  if (fZbeam) f << fAbeam << " " << fZbeam << " " << fAtarget << " " << fZtarget << " " << fEbeam << endl;

318  f << "Target: " << fTarget->NumberOfLayers() << " " << fTarget->GetAngleToBeam() << endl;
319  TIter next(fTarget->GetLayers());
320  KVMaterial* lay;
321  while ((lay = (KVMaterial*)next())) {
322  if (lay->IsIsotopic()) f << Form("%d%s", (Int_t)lay->GetMass(), lay->GetType());
323  else f << lay->GetType();
324  f << " " << lay->GetAreaDensity() / KVUnits::mg << endl;
325  }
326  }

328  GetRunList(runlist);
329  f << "Runs: " << runlist.AsString() << endl;

346 void KVDBSystem::Load(istream& f)
347 {
348  //Read and set informations on system in the format used in Systems.dat files:
349  //
350  //+155Gd + 238U 36 MeV/A '+' followed by name of system
351  //155 64 238 92 36.0 Aproj Zproj Atarg Ztarg Ebeam
352  //Target: 3 0.0 target with 3 layers, angle 0 degrees
353  //C 0.02 1st layer : carbon, 20 ug/cm2
354  //238U 0.1 2nd layer : uranium-238, 100 ug/cm2
355  //C 0.023 3rd layer : carbon, 23 ug/cm2
356  //Runs: 770-804 list of runs in KVNumberList format
357 
358  TString line;
359  Float_t target_thickness;
360  fAbeam = fZbeam = fAtarget = fZtarget = 0;
361  fEbeam = target_thickness = 0;
362  KVNumberList runlist;
363  //'peek' at first character of next line
364  char next_char = f.peek();
365  if (next_char == '+') {
366  line.ReadLine(f, kFALSE);
367  line.Remove(0, 1);
368  SetName(line.Data());
369  cout << "New System : " << line.Data() << endl;
370  } else {
371  Error("Load", "Should read system name : %s\n", line.Data());
372  return;
373  }
374  next_char = f.peek();
375  while (next_char != '+' && f.good() && !f.eof()) {
376  if ((next_char >= '0') && (next_char <= '9')) {
377  line.ReadLine(f, kFALSE);
378  if (sscanf(line.Data(), "%u %u %u %u %f %f", &fAbeam, &fZbeam, &fAtarget, &fZtarget, &target_thickness, &fEbeam) == 6) {
379  cout << "Zproj = " << fZbeam << " Ztarg = " << fZtarget << " targ_thick = " << target_thickness << " Ebeam = " << fEbeam << endl;
380  } else if (sscanf(line.Data(), "%u %u %u %u %f", &fAbeam, &fZbeam, &fAtarget, &fZtarget, &fEbeam) == 5) {
381  cout << "Zproj = " << fZbeam << " Ztarg = " << fZtarget << " Ebeam = " << fEbeam << endl;
382  }
383  } else {
384  line.ReadLine(f, kFALSE);
385  if (line.BeginsWith("Target")) {
386  fTarget = new KVTarget;
387  line.Remove(0, line.Index(":") + 1);
388  Int_t nlay;
389  Float_t angle;
390  sscanf(line.Data(), "%d %f", &nlay, &angle);
391  Char_t mat[10];
392  Float_t thick;
393  for (int i = 0; i < nlay; i++) {
394  line.ReadLine(f);
395  sscanf(line.Data(), "%s %f", mat, &thick);
396  fTarget->AddLayer(mat, thick);
397  }
398  fTarget->SetAngleToBeam(angle);
399  fTarget->Print();
400  } else if (line.BeginsWith("Runs")) {
401  line.Remove(0, line.Index(":") + 1);
402  runlist.SetList(line.Data());
403  cout << "Runs : " << line.Data() << endl;
404  } else if (line.BeginsWith("Run Range")) {
405  line.Remove(0, line.Index(":") + 1);
406  Int_t frun, lrun;
407  sscanf(line.Data(), "%i %i", &frun, &lrun);
408  runlist.Add(Form("%i-%i", frun, lrun));
409  cout << "Run range : " << line.Data() << endl;
410  }
411  }
412  next_char = f.peek();
413  }
414  if (runlist.GetNValues()) {
415  SetRuns(runlist);
416  }
417  //set target if not already done (old versions)
418  if (!fTarget && target_thickness > 0 && fZtarget > 0) {
419  KVNucleus n(fZtarget, fAtarget);
420  fTarget = new KVTarget(n.GetSymbol(), target_thickness);
421  fTarget->Print();
422  }
423 }

432 void KVDBSystem::SetRuns(KVNumberList& rl)
433 {
434  //Associate this system with the runs in the list
435  //Any previously associated runs are first removed (links in the runs will be removed too)
436  Info("SetRuns", "Setting runs for system %s : %s", GetName(), rl.AsString());
437  RemoveAllRuns();
438  rl.Begin();
439  KVDBTable* runtable = GetRunsTable();
440  Int_t run_number;
441  while (!rl.End()) {
442  run_number = rl.Next();
443  KVDBRun* run = (KVDBRun*)runtable->GetRecord(run_number);
444  if (run) {
445  if (run->GetSystem()) {
446  Error("SetRuns", "Associating run %d with system \"%s\" : run already associated with system \"%s\"",
447  run_number, GetName(), run->GetSystem()->GetName());
448  }
449  if (AddLink("Runs", run)) {
450  //use name of system as title of run
451  run->SetTitle(GetName());
452  } else {
453  Info("SetRuns", "Could not add link for run %d", run_number);
454  }
455  } else {
456  //Info("SetRuns", "Run %d not found in database", run_number);
457  }
458  }

469 void KVDBSystem::AddRun(KVDBRecord* rec)
470 {
471  //Associate the given run with this system.
472  //If the run was previously associated with another system, this association
473  //will be removed.
474  if (!rec) return;
475  if (!rec->InheritsFrom("KVDBRun")) {
476  Error("AddRun", "Called with pointer to an object of class %s; should inherit from KVDBRun!",
477  rec->ClassName());
478  return;
479  }
480  KVDBRun* run = (KVDBRun*)rec;
481  if (run->GetSystem()) run->GetSystem()->RemoveRun(run);
482  if (AddLink("Runs", run)) {
483  Info("AddRun", "Added link for run %d", run->GetNumber());
484  //use name of system as title of run
485  run->SetTitle(GetName());
486  } else {
487  Info("AddRun", "Could not add link for run %d", run->GetNumber());
488  }
489 }

592 KVDBTable* KVDBSystem::GetRunsTable()
593 {
594  //Deduce path to runs table in database from full path to parent table of this record.
595  //The systems are stored in a table called "Systems"
596  //The runs are stored in a table called "Runs"
597  //Therefore if we take the full path to the Systems table and replace Systems with Runs,
598  //we can then use gROOT->FindObject to get the pointer to the Runs table.
599 
600  TString path = fFullPathTable.Data();
601  path.ReplaceAll("Systems", "Runs");
602  return (KVDBTable*)gROOT->FindObject(path.Data());
603 }
