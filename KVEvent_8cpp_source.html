37 /*
38 <h2>KVEvent</h2>
39 <h4>Base class for all types of multiparticle event consisting of KVNucleus objects</h4>
40 
41 Particles are stored in a TClonesArray and KVEvent provides some basic
42 functionality for accessing and manipulating the list.
43 
44 Events can be built using any class derived from KVNucleus to represent particles.
45 These classes can allocate memory in their default ctor: when filling events in a loop,
46 the same 'particle' objects are re-used for each new event, the ctor of each object will
47 only be called once when the object is first created (e.g. in the first event).
48 The particle class Clear() method will be called before each new event.
49 Therefore the cycle of use of the particle objects in a loop over many events is:
50 
51 ~~~~~~~~~~~~~~~
52 <particle ctor>
53  Building 1st event
54  <particle Clear()>
55  Building 2nd event
56  <particle Clear()>
57  ...
58  Building last event
59 <particle dtor>
60 ~~~~~~~~~~~~~~~
61 
62 When writing events in a TTree, it is very important to call the TBranch::SetAutoDelete(kFALSE)
63 method of the branch which is used to store the event object.
64 
65 If not, when the events are read back, the KVEvent constructor and destructor will be called
66 every time an event is read from the TTree meading to very slow reading times (& probably
67 memory leaks).
68 
69 For this reason we provide the method:
70 
71 ~~~~~~~~~~~~~~~
72  void MakeEventBranch(TTree*, const TString&, const TString&, void*)
73 ~~~~~~~~~~~~~~~
74 
75 which should be used whenever it is required to stock KVEvent-derived objects in a TTree.
76 */

224 KVNucleus* KVEvent::AddParticle()
225 {
226  // Method used for building an event particle by particle.
227  // DO NOT USE FOR READING EVENTS - use GetParticle(Int_t npart)!!
228  //
229  // This method increases the multiplicity fMult by one and "creates" a new particle with index (fMult-1).
230  // In actual fact a new object is only created if needed i.e. if the new multiplicity is greater
231  // than previously. Particle objects in the array are reused from one event to another.
232  // The default constructor for the class corresponding to the "new" particle will only be called
233  // once during its lifetime (i.e. if N events are generated, the particle ctor will be called only once,
234  // not N times). Once created, in subsequent events we just call the particle's Clear() method
235  // in order to reset its internal variables ready for a new event.
236 
237  Int_t mult = GetMult();
238 #ifdef __WITHOUT_TCA_CONSTRUCTED_AT
239  KVNucleus* tmp = (KVNucleus*) ConstructedAt(mult, "C");
240 #else
241  KVNucleus* tmp = (KVNucleus*) fParticles->ConstructedAt(mult, "C");
242 #endif
243  if (!tmp) {
244  Error("AddParticle", "Allocation failure, Mult=%d", mult);
245  return 0;
246  }
247  return tmp;
248 }

345 {
346  //Returns multiplicity (number of particles) of event.
347  //If opt = "" (default), returns number of particles in TClonesArray* fParticles
348  // i.e. the value of fParticles->GetEntriesFast() (we assume there are no gaps
349  // in the list)
350  //If opt = "ok" only particles with IsOK()==kTRUE are included.
351  //If opt = "name" only particles with GetName()=="name" are included.

353  //IN THE LATTER TWO CASES, YOU MUST NOT USE THIS METHOD INSIDE A LOOP
354  //OVER THE EVENT USING GETNEXTPARTICLE() !!!

364  ResetGetNextParticle();

611 {
612  // Fill array mult[] with the number of each nuclear species in the
613  // comma-separated list in this event. Make sure that mult[] is

661 KVNucleus* KVEvent::GetNextParticle(Option_t* opt)
662 {
663  // Use this method to iterate over the list of particles in the event
664  // After the last particle GetNextParticle() returns a null pointer and
665  // resets itself ready for a new iteration over the particle list.
666  //
667  // If opt="" all particles are included in the iteration.
668  // If opt="ok" or "OK" only particles whose IsOK() method returns kTRUE are included.
669  //
670  // Any other value of opt is interpreted as a (case-insensitive) particle group name: only
671  // particles with BelongsToGroup(opt) returning kTRUE are included.
672  //
673  // If you want to start from the beginning again before getting to the end
674  // of the list, especially if you want to change the selection criteria,
675  // call method ResetGetNextParticle() before continuing.
676  // If you interrupt an iteration before the end, then start another iteration
677  // without calling ResetGetNextParticle(), even if you change the argument of
678  // the call to GetNextParticle(), you will repeat exactly the same iteration
679  // as the previous one.
680  //
681  // WARNING: Only one iteration at a time over the event can be performed
682  // using this method. If you want/need to perform several i.e. nested
683  // iterations, use the KVEvent::Iterator class
684 
685  TString Opt(opt);
686  Opt.ToUpper();
687 
688  // continue iteration
689  if (fIter.IsIterating()) return &(*(fIter++));
690 
691  // start new iteration
692  Bool_t ok_iter = (Opt == "OK");
693  Bool_t grp_iter = (!ok_iter && Opt.Length());
694 
695 #ifdef WITH_CPP11
696  if (ok_iter) fIter = Iterator(this, Iterator::Type::OK);
697  else if (grp_iter) fIter = Iterator(this, Iterator::Type::Group, Opt);
698  else fIter = Iterator(this, Iterator::Type::All);
699 #else
700  if (ok_iter) fIter = Iterator(this, Iterator::OK);
701  else if (grp_iter) fIter = Iterator(this, Iterator::Group, Opt);
702  else fIter = Iterator(this, Iterator::All);
703 #endif
704  return &(*(fIter++));
705 }

786 {

1326 Double_t KVEvent::GetChannelQValue() const
1327 {
1328  // Calculate the Q-value [MeV] for this event as if all nuclei were produced by
1329  // the decay of an initial compound nucleus containing the sum of all nuclei
1330  // in the event, i.e.
1331  // A -> a1 + a2 + a3 + ...
1332  // We take into account any excitation energy of the nuclei of the event
1333  // (see GetGSChannelQValue() for an alternative), i.e. we calculate
1334  // Q = M(A) - ( m(a1) + m(a2) + m(a3) + ... )
1335  // where
1336  // M(X) = ground state mass of X
1337  // m(X) = M(X) + E*(X)
1338  // If Q<0, the excitation energy of the initial compound nucleus, A,
1339  // would have to be at least equal to (-Q) in order for the decay to occur.
1340  // i.e. decay is possible if
1341  // E*(A) > -Q
1342 
1343  Double_t sumM = 0;
1344  KVNucleus CN;
1345  Int_t M = const_cast<KVEvent*>(this)->GetMult();
1346  for (int i = 1; i <= M; i++) {
1347  sumM += GetParticle(i)->GetMass();
1348  CN += *(GetParticle(i));
1349  }
1350  return CN.GetMassGS() - sumM;
1351 }
