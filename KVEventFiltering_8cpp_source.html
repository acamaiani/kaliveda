22 <h4>Filter simulated events with multidetector response</h4>

25 // Use this KVEventSelector on simulated data TTrees containing a branch with KVSimEvent-derived objects:
26 //
27 // my_tree->Process("KVEventFiltering", "[options]");
28 //
29 // where "[options]" is the list of options in the form "BranchName=toto,Dataset=titi,System=tata, ...".
30 //
31 // The following options MUST be given:
32 //
33 // SimFileName: name of file containing simulated events
34 // SimTitle: description of simulation
35 // BranchName: name of branch containing simulated events
36 // Dataset: name of experimental dataset (defines multidetector array to use etc.)
37 // System: name of experimental dataset system (defines collision kinematics etc.)
38 // Geometry: type of geometry, either 'KV' (KaliVeda geometry) or 'ROOT' (ROOT TGeometry package)
39 // Filter: type of filter, either 'Geo' (geometric filter), 'GeoThresh' (geometry + detector thresholds),
40 // or 'Full' (full simulation of detector response, including experimental identification
41 // and calibration routines)
42 // OutputDir: directory path in which to write filtered data file
43 // Kinematics: kinematical frame for simulation, either "cm" or "lab". if "cm", we use the c.m. velocity
44 // of the selected System to transform events into the detector (laboratory) frame.
45 // if "lab" no transformation is performed: simulated events are already in laboratory frame.
46 //
47 // The following are optional options:
48 //
49 // Run: run number to use for detector status, setup, parameters, etc.
50 // if not given, first run of the given experimental system is used.
51 // PhiRot: by default, a random rotation around the beam axis will be performed before
52 // simulating detection of the event. If you don't want this to happen,
53 // give option PhiRot=no
54 // Gemini: if option Gemini=yes, then each event will be "decayed" with Gemini++,
55 // if KaliVeda has been compiled with Gemini++ support.
56 //
57 // The filtered data will be written in the directory given as option "OutputDir".
58 // The filename is built up from the original simulation filename and the values
59 // of various options:
60 //
61 // [simfile]_geo=[geometry]_filt=[filter-type]_[dataset]_[system]_run=[run-number].root
62 //
63 // The data will be stored in a TTree with name 'ReconstructedEvents', in a branch with name
64 // 'ReconEvent'. The class used for reconstructed events depends on the dataset,
65 // it is given by KVDataSet::GetReconstructedEventClassName().

217 Bool_t KVEventFiltering::Analysis()
218 {
219  // Event-by-event filtering of simulated data.
220  // If needed (fTransformKinematics = kTRUE), kinematics of event are transformed
221  // to laboratory frame using C.M. velocity calculated in InitAnalysis().
222  // Detection of particles in event is simulated with KVMultiDetArray::DetectEvent,
223  // then the reconstructed detected event is treated by the same identification and calibration
224  // procedures as for experimental data.
225 
226  KVEvent* to_be_detected = GetEvent();
227  if (fGemini) {
228 #ifdef WITH_GEMINI
229  KVGemini g;
230  g.DecayEvent((KVSimEvent*)GetEvent(), &fGemEvent);
231  to_be_detected = &fGemEvent;
232 #endif
233  }
234  if (fTransformKinematics) {
235  if (fNewFrame == "proj") to_be_detected->SetFrame("lab", fProjVelocity);
236  else to_be_detected->SetFrame("lab", fCMVelocity);
237  if (fRotate) {
238  RandomRotation(to_be_detected, "lab");
239  gMultiDetArray->DetectEvent(to_be_detected, fReconEvent, "rotated_frame");
240  } else {
241  gMultiDetArray->DetectEvent(to_be_detected, fReconEvent, "lab");
242  }
243  } else {
244  if (fRotate) {
245  RandomRotation(to_be_detected);
246  gMultiDetArray->DetectEvent(to_be_detected, fReconEvent, "rotated_frame");
247  } else {
248  gMultiDetArray->DetectEvent(to_be_detected, fReconEvent);
249  }
250  }
251  fReconEvent->SetNumber(fEVN++);
252  fReconEvent->SetFrameName("lab");
253  fTree->Fill();
254 
255  /* if (!(fEventsRead % fEventsReadInterval) && fEventsRead) {
256  memory_check.Check();
257  }
258  */
259  return kTRUE;
260 }

296 void KVEventFiltering::InitAnalysis()
297 {
298  // Select required dataset for filtering (option "Dataset")
299  // Build the associated multidetector geometry.
300  // Calculate C.M. velocity associated with required experimental collision
301  // kinematics (option "System").
302  //
303  // Set the parameters of the detectors according to the required run
304  // if given (option "Run"), or the first run of the given system otherwise.
305  // If ROOT/TGeo geometry is required (option "Geometry"="ROOT"),
306  // build the TGeometry representation of the detector array.
307  //
308  // Open file for filtered data (see KVEventFiltering::OpenOutputFile), which will
309  // be stored in a TTree with name 'ReconstructedEvents', in a branch with name
310  // 'ReconEvent'. The class used for reconstructed events depends on the dataset,
311  // it is given by KVDataSet::GetReconstructedEventClassName().
312 
313  if (fDisableCreateTreeFile) return; //when running with PROOF, only execute on workers
314 
315  TString dataset = GetOpt("Dataset").Data();
316  if (!gDataSetManager) {
317  gDataSetManager = new KVDataSetManager;
318  gDataSetManager->Init();
319  }
320  gDataSetManager->GetDataSet(dataset)->cd();
321 
322  TString system = GetOpt("System").Data();
323  KVDBSystem* sys = (gDataBase ?
324  (gDataBase->GetTable("Systems") ? (KVDBSystem*)gDataBase->GetTable("Systems")->GetRecord(system) : nullptr)
325  : nullptr);
326  KV2Body* tb = 0;
327 
328  Bool_t justcreated = kFALSE;
329  if (sys) tb = sys->GetKinematics();
330  else if (system) {
331  tb = new KV2Body(system);
332  tb->CalculateKinematics();
333  justcreated = kTRUE;
334  }
335 
336  fCMVelocity = (tb ? tb->GetCMVelocity() : TVector3(0, 0, 0));
337  fCMVelocity *= -1.0;
338 
339  fProjVelocity = (tb ? tb->GetNucleus(1)->GetVelocity() : TVector3(0, 0, 0));
340  fProjVelocity *= -1.0;
341 
342  if (justcreated)
343  delete tb;
344 
345  Int_t run = 0;
346  if (IsOptGiven("Run")) {
347  run = GetOpt("Run").Atoi();
348  Info("InitAnalysis", "Run given in options = %d", run);
349  }
350  if (!run) {
351  if (sys) {
352  run = ((KVDBRun*)sys->GetRuns()->First())->GetNumber();
353  Info("InitAnalysis", "Using first run for system = %d", run);
354  } else {
355  Info("InitAnalysis", "No run information");
356  run = -1;
357  }
358  }
359 
360  KVMultiDetArray::MakeMultiDetector(dataset, run);
361  if (run == -1) gMultiDetArray->InitializeIDTelescopes();
362  gMultiDetArray->SetSimMode();
363 
364  TString geo = GetOpt("Geometry").Data();
365  if (geo == "ROOT") {
366  gMultiDetArray->CheckROOTGeometry();
367  Info("InitAnalysis", "Filtering with ROOT geometry");
368  Info("InitAnalysis", "Navigator detector name format = %s", gMultiDetArray->GetNavigator()->GetDetectorNameFormat());
369  } else {
370  gMultiDetArray->SetROOTGeometry(kFALSE);
371  Info("InitAnalysis", "Filtering with KaliVeda geometry");
372  }
373 
374  TString filt = GetOpt("Filter").Data();
375  if (filt == "Geo") {
376  gMultiDetArray->SetFilterType(KVMultiDetArray::kFilterType_Geo);
377  Info("InitAnalysis", "Geometric filter");
378  } else if (filt == "GeoThresh") {
379  gMultiDetArray->SetFilterType(KVMultiDetArray::kFilterType_GeoThresh);
380  Info("InitAnalysis", "Geometry + thresholds filter");
381  } else if (filt == "Full") {
382  gMultiDetArray->SetFilterType(KVMultiDetArray::kFilterType_Full);
383  Info("InitAnalysis", "Full simulation of detector response & calibration");
384  }
385  TString kine = GetOpt("Kinematics").Data();
386  if (kine == "lab") {
387  fTransformKinematics = kFALSE;
388  Info("InitAnalysis", "Simulation is in laboratory/detector frame");
389  } else {
390  fNewFrame = kine;
391  Info("InitAnalysis", "Simulation will be transformed to laboratory/detector frame");
392  }
393 
394  if (IsOptGiven("PhiRot")) {
395  if (GetOpt("PhiRot") == "no") fRotate = kFALSE;
396  }
397  if (fRotate) Info("InitAnalysis", "Random phi rotation around beam axis performed for each event");
398 #ifdef WITH_GEMINI
399  if (IsOptGiven("Gemini")) {
400  if (GetOpt("Gemini") == "yes") fGemini = kTRUE;
401  }
402  if (fGemini) Info("InitAnalysis", "Statistical decay with Gemini++ for each event before detection");
403 #endif
404  if (!gDataSet->HasCalibIdentInfos()) {
405  TString fullpath;
406  if (KVBase::SearchKVFile("IdentificationBilan.dat", fullpath, gDataSet->GetName())) {
407  Info("InitAnalysis", "Setting identification bilan...");
408  TString sysname = sys->GetBatchName();
409  KVIDTelescope::OpenIdentificationBilan(fullpath);
410  TIter next(gMultiDetArray->GetListOfIDTelescopes());
411  KVIDTelescope* idt;
412  while ((idt = (KVIDTelescope*)next())) {
413  idt->CheckIdentificationBilan(sysname);
414  }
415  }
416  }
417  gMultiDetArray->PrintStatusOfIDTelescopes();
418 
419  OpenOutputFile(sys, run);
420  if (sys) fTree = new TTree("ReconstructedEvents", Form("%s filtered with %s (%s)", GetOpt("SimTitle").Data(), gMultiDetArray->GetTitle(), sys->GetName()));
421  else fTree = new TTree("ReconstructedEvents", Form("%s filtered with %s", GetOpt("SimTitle").Data(), gMultiDetArray->GetTitle()));
422 
423  TString reconevclass = gDataSet->GetReconstructedEventClassName();
424  fReconEvent = (KVReconstructedEvent*)TClass::GetClass(reconevclass)->New();
425  KVEvent::MakeEventBranch(fTree, "ReconEvent", reconevclass, &fReconEvent);
426 
427  AddTree(fTree);
428 }

464 void KVEventFiltering::OpenOutputFile(KVDBSystem* S, Int_t run)
465 {
466  // Open ROOT file for new filtered events TTree.
467  // The file will be written in the directory given as option "OutputDir".
468  // The filename is built up from the original simulation filename and the values
469  // of various options:
470  //
471  // [simfile]_[Gemini]_geo=[geometry]_filt=[filter-type]_[dataset]_[system]_run=[run-number].root
472  //
473  // In addition, informations on the filtered data are stored in the file as
474  // TNamed objects. These can be read by KVSimDir::AnalyseFile:
475  //
476  // KEY: TNamed System;1 title=[full system name]
477  // KEY: TNamed Dataset;1 title=[dataset name]
478  // KEY: TNamed Run;1 title=[run-number]
479  // KEY: TNamed Geometry;1 title=[geometry-type]
480  // KEY: TNamed Filter;1 title=[filter-type]
481  // KEY: TNamed Origin;1 title=[name of simulation file]
482  // KEY: TNamed RandomPhi;1 title=[yes/no, random rotation about beam axis]
483  // KEY: TNamed Gemini++;1 title=[yes/no, Gemini++ decay before detection]
484  //
485  TString basefile = GetOpt("SimFileName");
486  basefile.Remove(basefile.Index(".root"), 5);
487  TString outfile = basefile;
488  if (fGemini) outfile += "_Gemini";
489  outfile += "_geo=";
490  outfile += GetOpt("Geometry");
491  outfile += "_filt=";
492  outfile += GetOpt("Filter");
493  outfile += "_";
494  outfile += gDataSet->GetName();
495 
496  if (S) {
497  outfile += "_";
498  outfile += S->GetBatchName();
499  outfile += "_run=";
500  outfile += Form("%d", run);
501  } else if (GetOpt("System")) {
502  TString tmp = GetOpt("System");
503  tmp.ReplaceAll(" ", "");
504  tmp.ReplaceAll("/", "");
505  tmp.ReplaceAll("@", "");
506  tmp.ReplaceAll("MeV", "");
507  tmp.ReplaceAll("A", "");
508  tmp.ReplaceAll("+", "");
509  outfile += "_";
510  outfile += tmp.Data();
511  outfile += "_run=0";
512  }
513  outfile += ".root";
514 
515  TString fullpath;
516  AssignAndDelete(fullpath, gSystem->ConcatFileName(GetOpt("OutputDir").Data(), outfile.Data()));
517 
518  //fFile = new TFile(fullpath,"recreate");
519  CreateTreeFile(fullpath);
520 
521  TDirectory* curdir = gDirectory;
522  writeFile->cd();
523  if (S) {
524  TNamed* system = new TNamed("System", S->GetName());
525  system->Write();
526  } else if (GetOpt("System"))(new TNamed("System", GetOpt("System").Data()))->Write();
527 
528  (new TNamed("Dataset", gDataSet->GetName()))->Write();
529  if (S)(new TNamed("Run", Form("%d", run)))->Write();
530  if (gMultiDetArray->IsROOTGeometry()) {
531  (new TNamed("Geometry", "ROOT"))->Write();
532  } else {
533  (new TNamed("Geometry", "KV"))->Write();
534  }
535  (new TNamed("Filter", GetOpt("Filter").Data()))->Write();
536  (new TNamed("Origin", (basefile + ".root").Data()))->Write();
537  (new TNamed("RandomPhi", (fRotate ? "yes" : "no")))->Write();
538  (new TNamed("Gemini++", (fGemini ? "yes" : "no")))->Write();
539  curdir->cd();
540 }
