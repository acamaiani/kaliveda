80 const Char_t* KVIDINDRACsI::GetArrayName()
81 {
82  // Name of telescope given in the form CSI_R_L_Ring-numberTelescope-number
83  //where ring and telescope numbers are those of the smallest (in angular terms)
84  //detector of the telescope (if both are the same size, either one will do).
85  // The detectors are signified by their TYPE names i.e. KVDetector::GetType
86 
87  //in order to access angular dimensions of detectors, we need their KVTelescopes
88  KVINDRATelescope* de_det = dynamic_cast<KVINDRADetector*>(GetDetector(1))->GetTelescope();
89  UInt_t ring, mod;
90  ring = de_det->GetRingNumber();
91  mod = de_det->GetNumber();
92  TString dummy;
93  dummy.Form("%s_R_L", GetDetector(1)->GetType());
94  SetType(dummy.Data());
95  dummy.Form("%s_%02d%02d", GetType(), ring, mod);
96  SetName(dummy.Data());
97 
98  return fName.Data();
99 }

107 Bool_t KVIDINDRACsI::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)
108 {
109  //Particle identification and code setting using identification grid KVIDGCsI* fGrid.
110 
111  IDR->SetIDType(GetType());
112  IDR->IDattempted = kTRUE;
113 
114  //perform identification
115  Double_t csir = (y < 0. ? GetIDMapY() : y);
116  Double_t csil = (x < 0. ? GetIDMapX() : x);
117  CsIGrid->Identify(csil, csir, IDR);
118 
119  // set general ID code
120  IDR->IDcode = kIDCode2;
121 
122  // general ID code for gammas
123  if (IDR->IDquality == KVIDGCsI::kICODE10)
124  IDR->IDcode = kIDCode0;
125 
126  return kTRUE;
127 
128 }

165 void KVIDINDRACsI::Initialize()
166 {
167  // Initialisation of telescope before identification.
168  // This method MUST be called once before any identification is attempted.
169  // Initialisation of grid is performed here.

176  CsIGrid->Initialize();
177  SetBit(kReadyForID);
