150 KVIonRangeTableMaterial* KVRangeYanez::GetMaterialWithNameOrType(const Char_t* material)
151 {
152  // Returns pointer to material of given name or type.
153  // Note that any request for an element of the periodic table will cause
154  // the corresponding material to be created (if not already done).
155  // If given in the form of an isotope ("48Ca", "124Sn", etc.) this material will
156  // be isotopically pure. If not ("Ca", "natSn", etc.) the material generated will be
157  // a mixture of the most abundant naturally-occurring isotopes.
158 
159  CheckMaterialsList();
160  KVIonRangeTableMaterial* M = (KVIonRangeTableMaterial*)fMaterials->FindObject(material);
161  if (!M) {
162  M = (KVIonRangeTableMaterial*)fMaterials->FindObjectByType(material);
163  }
164  if (!M) {
165  // is the requested material an atomic element?
166  Int_t A = KVNucleus::IsMassGiven(material);
167  if (A > -1) {
168  KVNucleus n(material);
169  AddElementalMaterial(n.GetZ(), A);
170  M = (KVIonRangeTableMaterial*)fMaterials->FindObjectByType(material);
171  }
172  }
173  return M;
174 }

238 void KVRangeYanez::AddElementalMaterial(Int_t z, Int_t a)
239 {
240  // Adds a material composed of a single isotope of a chemical element.
241  // If the isotope (a) is not specified, we create a material containing the naturally
242  // occuring isotopes of the given element, weighted according to their abundance.
243  // If the mass is given, the material symbol will be "AX" where X is the symbol for the element
244  // e.g. "48Ca", "124Sn", etc.
245  // and the material name will be "Xxx-A" where Xxx is the name of the element
246  // e.g. "Calcium-48", "Tin-124", etc.
247  // Otherwise, we just use the element symbol and name for naturally-occurring
248  // mixtures of atomic elements ("Ca", "Calcium", etc.).
249 
250  KVIonRangeTableMaterial* mat;
251  if (!a) {
252  mat = MakeNaturallyOccuringElementMixture(z);
253  } else {
254  if (!gNDTManager) {
255  Error("AddElementalMaterial",
256  "Nuclear data tables have not been initialised");
257  return;
258  }
259  KVElementDensity* ed = (KVElementDensity*)gNDTManager->GetData(z, a, "ElementDensity");
260  if (!ed) {
261  Error("AddElementalMaterial",
262  "No element found in ElementDensity NDT-table with Z=%d", z);
263  return;
264  }
265  TString state = "solid";
266  if (ed->IsGas()) state = "gas";
267  mat = new KVRangeYanezMaterial(this, Form("%s-%d", ed->GetElementName(), a),
268  Form("%d%s", a, ed->GetElementSymbol()),
269  state, ed->GetValue(), z, a);
270  mat->Initialize();
271  }
272  CheckMaterialsList();
273  fMaterials->Add(mat);
274  mat->ls();
275 }

340 KVIonRangeTableMaterial* KVRangeYanez::MakeNaturallyOccuringElementMixture(Int_t z)
341 {
342  // create a material containing the naturally occuring isotopes of the given element,
343  // weighted according to their abundance.
344 
345  if (!gNDTManager) {
346  Error("MakeNaturallyOccuringElementMixture",
347  "Nuclear data tables have not been initialised");
348  return NULL;
349  }
350  KVElementDensity* ed = (KVElementDensity*)gNDTManager->GetData(z, z, "ElementDensity");
351  if (!ed) {
352  Error("AddElementalMaterial",
353  "No element found in ElementDensity NDT-table with Z=%d", z);
354  return 0x0;
355  }
356  TString state = "solid";
357  if (ed->IsGas()) state = "gas";
358  KVRangeYanezMaterial* mat =
359  new KVRangeYanezMaterial(this,
360  ed->GetElementName(),
361  ed->GetElementSymbol(),
362  state, ed->GetValue());
363  KVNucleus nuc(z);
364  KVNumberList isotopes = nuc.GetKnownARange();
365  isotopes.Begin();
366  while (!isotopes.End()) {
367 
368  nuc.SetA(isotopes.Next());
369  Double_t abundance = nuc.GetAbundance() / 100.;
370  if (abundance > 0.) mat->AddMixtureElement(z, nuc.GetA(), 1, abundance);
371 
372  }
373  mat->Initialize();
374  return (KVIonRangeTableMaterial*)mat;
375 }

382 void KVRangeYanez::ReadPredefinedMaterials(const Char_t* filename)
383 {
384  // Read materials from file whose name is given
385 
386  TString DataFilePath = filename;
387 
388  ifstream filestream;
389  if (!SearchAndOpenKVFile(DataFilePath, filestream, "data")) {
390  Error("ReadPredefinedMaterials", "Cannot open %s for reading", DataFilePath.Data());
391  return;
392  }
393  Info("ReadPredefinedMaterials", "Reading materials in file : %s", filename);
394 
395  Bool_t compound, mixture;
396  compound = mixture = kFALSE;
397 
398  KVString line;
399  while (filestream.good()) {
400  line.ReadLine(filestream);
401  if (filestream.good()) {
402  if (line.BeginsWith("//")) continue;
403  if (line.BeginsWith("COMPOUND")) {
404  compound = kTRUE;
405  mixture = kFALSE;
406  } else if (line.BeginsWith("MIXTURE")) {
407  compound = kFALSE;
408  mixture = kTRUE;
409  }
410  if (compound || mixture) {
411  // new compound or mixed material
412  KVString name, symbol, state;
413  Double_t density = -1;
414  KVString element[10];
415  Int_t natoms[10];
416  Int_t z[10], a[10];
417  Double_t proportion[10];
418  Int_t nelem = 0;
419  line.ReadLine(filestream);
420  while (filestream.good() && !line.IsWhitespace() && line != "\n") {
421  line.Begin("=");
422  KVString next = line.Next();
423  if (next == "name") name = line.Next();
424  else if (next == "symbol") symbol = line.Next();
425  else if (next == "state") state = line.Next();
426  else if (next == "density") density = line.Next().Atof();
427  else if (next == "nelem") {
428  nelem = line.Next().Atoi();
429  for (int i = 0; i < nelem; i++) {
430  line.ReadLine(filestream);
431  line.Begin(" ");
432  element[i] = line.Next();
433  a[i] = KVNucleus::IsMassGiven(element[i]);
434  KVNucleus n(element[i]);
435  z[i] = n.GetZ();
436  if (!a[i]) a[i] = TMath::Nint(n.GetNaturalA());
437  natoms[i] = line.Next().Atoi();
438  if (mixture) proportion[i] = line.Next().Atof();
439  }
440  }
441  line.ReadLine(filestream, kFALSE); //do not skip 'whitespace'
442  }
443  if (compound) AddCompoundMaterial(name, symbol, nelem, z, a, natoms, density);
444  else if (mixture) AddMixedMaterial(name, symbol, nelem, z, a, natoms, proportion, density);
445  compound = mixture = kFALSE;
446  }
447  }
448  }
449 }
